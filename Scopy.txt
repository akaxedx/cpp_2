#include<string>
#include<fstream>
#include<iostream>
using namespace std;

int main() {
	/*写入txt文件*/
	ifstream in("Scopy.cpp");
	ofstream out("Scopy.txt");
	string s;
	while (getline(in,s))//逐行读取字符遇到换行符退出
	{
		out << s << "\n";//写入换行符
	}
	/*写入字符串*/
	ifstream in2("Scopy.cpp");
	string s2, line;
	while (getline(in2,line))
	{
		s2 += line + "\n";
	}
	cout << s2;
}
